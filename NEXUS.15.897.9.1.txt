## INDEXING

# EACH NODE IN THE
# CHAIN ARE GIVEN
# A SPECIFIC
# INDEX. THE
# INDICIES GOES
# FROM 200 IN
# REVERSE ORDER.
# SO 200 IS THE
# NEWEST VALUE,
# 199 IS THE
# SECOND NEWEST,
# AND SO ON.

@1
   MOV UP ACC
   ADD 200
   MOV ACC RIGHT

@2
   MOV UP ACC
   SAV
   MOV LEFT ACC
   SUB 200
   JNZ Z
   SWP
   MOV ACC RIGHT
   JMP E
Z: ADD 200
   MOV ACC RIGHT
   SWP
E: MOV ACC RIGHT

@3

   MOV LEFT ACC
   SUB 199
   JNZ Z
   SWP
   MOV ACC DOWN
   JMP E
Z: ADD 199
   MOV ACC DOWN
   SWP
E: MOV ACC DOWN
   MOV LEFT ACC
